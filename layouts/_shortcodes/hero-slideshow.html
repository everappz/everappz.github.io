{{- /* Hero slideshow: cycles through product hero images with fade transitions. Each slide shows the hero screenshot, app icon + name (bottom-left), and Read more button (bottom-right). Pulls data from product pages automatically. Usage: hero-slideshow (no parameters needed) */ -}} {{- $products := slice -}} {{- $productOrder := slice "evermusic" "flacbox" "evervideo" "evertag" -}} {{- range $slug := $productOrder -}} {{- $page := site.GetPage (printf "/products/%s" $slug) -}} {{- if $page -}} {{- $products = $products | append $page -}} {{- end -}} {{- end -}} {{- $slides := slice -}} {{- range $idx, $prod := $products -}} {{- $heroImage := "" -}} {{- $icon := "" -}} {{- $name := $prod.Title -}} {{- $link := $prod.RelPermalink -}} {{- $appStoreUrl := $prod.Params.appStoreUrl | default "" -}} {{- $slug := index $productOrder $idx -}} {{- $heroPath := printf "products/%s/heroimage/hero_1600.webp" $slug -}} {{- with resources.Get $heroPath -}} {{- $processed := .Process "Resize 1200x webp q80" -}} {{- $heroImage = $processed.RelPermalink -}} {{- else -}} {{- $heroPath2 := printf "products/%s/heroimage/hero_1200.webp" $slug -}} {{- with resources.Get $heroPath2 -}} {{- $processed := .Process "Resize 1200x webp q80" -}} {{- $heroImage = $processed.RelPermalink -}} {{- end -}} {{- end -}} {{- $iconBaseName := "" -}} {{- if eq $slug "evermusic" }}{{ $iconBaseName = "Evermusic" }} {{- else if eq $slug "flacbox" }}{{ $iconBaseName = "Flacbox" }} {{- else if eq $slug "evervideo" }}{{ $iconBaseName = "Evervideo" }} {{- else if eq $slug "evertag" }}{{ $iconBaseName = "Evertag" }} {{- end -}} {{- $icon = printf "/images/app_icons/webp/%s_Icon-App-1024x1024.webp" $iconBaseName -}} {{- if $heroImage -}} {{- $slide := dict "image" $heroImage "icon" $icon "name" $name "link" $link "appStoreUrl" $appStoreUrl -}} {{- $slides = $slides | append $slide -}} {{- end -}} {{- end -}} {{- $slideshowID := printf "hero-slideshow-%d" now.UnixNano -}}
{{- range $i, $slide := $slides }}
{{- if $slide.appStoreUrl }} {{ $slide.name }} {{- else }}
{{ $slide.name }}
{{- end }} {{- if $slide.link }} Read more {{- end }}
{{- end }}
{{- if gt (len $slides) 1 }}
{{- range $i, $slide := $slides }} {{- end }}
{{- end }}
{{- if gt (len $slides) 1 }} {{- end }}
