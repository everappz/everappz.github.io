{{- /* i18n-aware card wrapper. Accepts same params as card, but any param can also be an i18n key that gets translated via T. If a value starts with "i18n:" it is treated as a translation key. Otherwise it is used as a literal string (pass-through to card). @example {{< i18n-card link="/docs/guide" title="i18n:cardUserGuide" subtitle="i18n:cardUserGuideDesc" icon="book-open" >}} @example {{< i18n-card link="/about" title="About" >}} */ -}} {{- $title := .Get "title" | default "" -}} {{- $subtitle := .Get "subtitle" | default "" -}} {{- $tag := .Get "tag" | default "" -}} {{- if hasPrefix $title "i18n:" }}{{ $title = T (strings.TrimPrefix "i18n:" $title) }}{{ end -}} {{- if hasPrefix $subtitle "i18n:" }}{{ $subtitle = T (strings.TrimPrefix "i18n:" $subtitle) }}{{ end -}} {{- if hasPrefix $tag "i18n:" }}{{ $tag = T (strings.TrimPrefix "i18n:" $tag) }}{{ end -}} {{- $link := .Get "link" -}} {{- $icon := .Get "icon" -}} {{- $image := .Get "image" -}} {{- $alt := .Get "alt" | default $title -}} {{- $width := 0 -}} {{- $height := 0 -}} {{- $imageStyle := .Get "imageStyle" -}} {{- $tagColor := .Get "tagColor" | default (.Get "tagType") | default "" -}} {{- $tagBorder := not (eq (.Get "tagBorder") false) | default true }} {{- $tagIcon := .Get "tagIcon" | default "" -}} {{- $method := .Get "method" | default "Resize" | humanize -}} {{- $options := .Get "options" | default "800x webp q80" -}} {{- $process := .Get "process" | default (printf "%s %s" $method $options) -}} {{- if and $image (not (urls.Parse $image).Scheme) -}} {{- with or (.Page.Resources.Get $image) (resources.Get $image) -}} {{- if (not (eq .MediaType.SubType "svg")) -}} {{- $processed := .Process $process -}} {{- $width = $processed.Width -}} {{- $height = $processed.Height -}} {{- $image = $processed.RelPermalink -}} {{- end -}} {{ else }} {{- if hasPrefix $image "/" -}} {{- $image = relURL (strings.TrimPrefix "/" $image) -}} {{- end -}} {{- end -}} {{- end -}} {{- partial "shortcodes/card" (dict "page" .Page "link" $link "title" $title "icon" $icon "subtitle" $subtitle "image" $image "alt" $alt "width" $width "height" $height "imageStyle" $imageStyle "tag" $tag "tagType" $tagColor "tagBorder" $tagBorder "tagIcon" $tagIcon ) -}}
